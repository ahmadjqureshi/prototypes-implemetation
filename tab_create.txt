CREATE TABLE AccountInfo(
			Account varchar2(12) , 
			AccountTitle varchar2(256) , 
			CapturedDate varchar2(20) , 
			CompanyName varchar2(256) , 
			AccountDetails varchar2(1000), 
			AccountStatus varchar2(25) Not Null, 
			CapturedBy varchar(256),
			LastModifiedByIP varchar2(15),
			LastVerifiedByIP varchar2(15),
			Constraint Account_PK Primary Key(Account)
			);
